Getting Started

Moon Animator 2 exposes its internal API through a global table, allowing plugin developers to extend functionality, create custom tools, and automate animation workflows.

Unofficial DocumentationThis API is not officially documented. All data was extracted via runtime introspection. Structure may change between versions.

Access Point

Lua
local _g

-- Wait for Moon Animator to fully load
repeat
    _g = _G.MoonGlobal
    task.wait(0.5)
until _g and _g.ready

print("Moon Animator v", _g.ver) -- 34000

Plugin Modding

Save as Local Plugin in Roblox Studio (right-click → "Save as Local Plugin…").

Always wait: repeat task.wait(0.5) until _G.MoonGlobal and _G.MoonGlobal.ready

Custom Easings

CustomEasings.lua
-- Save as Local Plugin
local EASINGS = {
    { name="Soft", color=Color3.fromRGB(86,180,233), power=1.5 },
    { name="Tert", color=Color3.fromRGB(230,159,0),  power=2.5 },
}

local function makeIn(p)    return function(t) return t^p end end
local function makeOut(p)   return function(t) return 1-(1-t)^p end end
local function makeInOut(p) return function(t)
    if t < .5 then return ((2*t)^p)/2
    else return 1-((-2*t+2)^p)/2 end end end
local function makeOutIn(p) return function(t)
    if t < .5 then return (1-(1-2*t)^p)/2
    else return ((2*t-1)^p+1)/2 end end end

local _g
for i=1,60 do
    _g=_G.MoonGlobal
    if _g and _g.EasingFunctions and _g.class then break end
    if i==60 then return end
    task.wait(.5)
end
local Ease=require(_g.class.Ease)

for _,e in ipairs(EASINGS) do
    _g.EasingFunctions[e.name.."In"]=makeIn(e.power)
    _g.EasingFunctions[e.name.."Out"]=makeOut(e.power)
    _g.EasingFunctions[e.name.."InOut"]=makeInOut(e.power)
    _g.EasingFunctions[e.name.."OutIn"]=makeOutIn(e.power)
    Ease.EASE_DATA[e.name]={Color=e.color,Params={Direction=true}}
    table.insert(Ease.EASE_LIST,{e.name,e.name})
end
print("[CustomEasings]",#EASINGS,"loaded")

Theme Unlocker

ThemeUnlocker.lua
-- Save as Local Plugin
local _g
for i=1,60 do
    _g=_G.MoonGlobal
    if _g and _g.Themer and _g.AllMenuItems then break end
    if i==60 then return end
    task.wait(.5)
end
task.wait(1)

local T=_g.Themer
local PAID={"Taiga","Blush","Blonde","Wheat","Gaia","Ares","Aqua","Energy","Digital","Fuchsia"}

local function applyTheme(name)
    local src=T._BuiltInThemes[name]
    if not src then return false end
    for k,v in pairs(src) do
        if typeof(v)=="table" and typeof(v[1])=="number"
           and typeof(v[2])=="number" and typeof(v[3])=="number" then
            rawset(T._Theme,k,Color3.fromRGB(v[1],v[2],v[3]))
        end end
    pcall(function() T:_paintAll() end)
    pcall(function() _g.plugin:SetSetting(_g.theme_key,name) end)
    return true
end

for _,name in ipairs(PAID) do
    for _,pfx in ipairs({"Theme_","Activate_Theme_"}) do
        local item=_g.AllMenuItems[pfx..name]
        if not item then continue end
        if item.price_frame then item.price_frame.Visible=false end
        local inp=item.middle_ui and item.middle_ui:FindFirstChild("Input")
        if inp and inp:IsA("GuiButton") then
            inp.MouseButton1Click:Connect(function() applyTheme(name) end)
        end end end
print("[ThemeUnlocker]",#PAID,"unlocked")
